As regras de validação são definidas dentro do arquivo "actions/fortify/CreateNewUser.php"
nesse arquivo existe a função de criação de usuario e antes de criar ela faz essa validação.
aqui eu posso inserir mais regras se eu quiser 

 public function create(array $input): User
    {
        Validator::make($input, [
            'name' => ['required', 'string', 'max:255'],
            'email' => [
                'required',
                'string',
                'email',
                'max:255',
                Rule::unique(User::class),
            ],
            'password' => $this->passwordRules(),
        ])->validate();

        return User::create([
            'name' => $input['name'],
            'email' => $input['email'],
            'password' => Hash::make($input['password']),
        ]);
    }
